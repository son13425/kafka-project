#  Команда создания топика с 3 партициями и 2 репликами через консоль
docker exec -it kafka-kraft-1 kafka-topics.sh \
  --create \
  --topic messages_from_frontend \
  --partitions 3 \
  --replication-factor 2 \
  --bootstrap-server localhost:9092

Параметры:
--topic — имя топика (желаемое название, здесь - messages_from_frontend)
--partitions 3 — 3 партиции
--replication-factor 2 — 2 реплики на каждую партицию
--bootstrap-server localhost:9092 — брокер кластера кафка

Вывод: Created topic messages_from_frontend.


#  Команда-проверка создания топика
docker exec -it kafka-kraft-1 kafka-topics.sh \
  --list \
  --bootstrap-server localhost:9092

Вывод: messages_from_frontend
